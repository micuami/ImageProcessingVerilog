`timescale 1ns / 1ps

module process(
	input clk,				// clock 
	input [23:0] in_pix,	// valoarea pixelului de pe pozitia [in_row, in_col] din imaginea de intrare (R 23:16; G 15:8; B 7:0)
	output reg [5:0] row, col, 	// selecteaza un rand si o coloana din imagine
	output reg out_we, 			// activeaza scrierea pentru imaginea de iesire (write enable)
	output reg [23:0] out_pix,	// valoarea pixelului care va fi scrisa in imaginea de iesire pe pozitia [out_row, out_col] (R 23:16; G 15:8; B 7:0)
	output reg mirror_done,		// semnaleaza terminarea actiunii de oglindire (activ pe 1)
	output reg gray_done,		// semnaleaza terminarea actiunii de transformare in grayscale (activ pe 1)
	output reg filter_done);	// semnaleaza terminarea actiunii de aplicare a filtrului de sharpness (activ pe 1)

// TODO add your finite state machines here
reg [5:0] state, next_state = 0;
reg [5:0] next_row, next_col = 0;
reg [23:0] pixel_aux1, pixel_aux2 = 0; //in ele pastrez valorile pixelilor pentru oglindire
reg [7:0] r, g, b; //in ele extrag culorile din fiecare in_pix
reg [10:0] old_pix, new_pix = 0; //in ele calculez noul pixel la sharpen

//partea secventiala
	 always @(posedge clk) begin
		state <= next_state;
		col <= next_col;
		row <= next_row;
		old_pix <= new_pix;
	 end

//partea combinationala
	 always @(*) begin
			 case(state)
					 0: begin //incep mirror cu initializarile sale
						next_row = 0;
						next_col = 0;
						mirror_done = 0;
						out_we = 0;
						next_state = 1;
					 end
					 1: begin //selectez pixelul curent si il pastrez in variabila auxiliara 1
						pixel_aux1 = in_pix;
						next_state = 2;
					 end
					 2: begin
						if(row < 32) begin //pentru pixelii din jumatatea superioara a matricei aleg oglinditul
							next_row = 63 - row;
							next_state = 3;
						end
						else begin //daca am trecut de jumatate inseamna ca am terminat de aplicat mirror
							mirror_done = 1;
							next_state = 7;
						end
					 end
					 3: begin //selectez pixelul curent si il pastrez in variabila auxiliara 2
						out_we = 1;
						pixel_aux2 = in_pix;
						out_pix = pixel_aux1; //pe cel din variabila auxiliara 1 il pun pe iesire
						next_state = 4;
					 end
					 4: begin //ma intorc inapoi in pixelul din care am plecat
						out_we = 0;
						next_row = 63 - row;
						next_state = 5;
					 end
					 5: begin //pun ce am in variabila auxiliara 2 pe iesire
						out_we = 1;
						out_pix = pixel_aux2;
						next_state = 6;
					 end
					 6: begin //trec la urmatorul pixel si repet procedura
					   out_we = 0;
						if(col < 63) begin
							next_col = col + 1;
						end
						else begin
							next_row = row + 1;
							next_col = 0;
						end
						next_state = 1;
					 end
					 7: begin //incep grayscale cu initializarile sale
						next_row = 0;
						next_col = 0;
						gray_done = 0;
						out_we = 0;
						next_state = 8;
					 end
					 8: begin //aplic grayscale punand noua valoare in out_pix[15:8] (la jumatate)
						b = in_pix[7:0];
						g = in_pix[15:8];
						r = in_pix[23:16];
						out_pix = 0;
						if((b <= g && g <= r) || (r <= g && g <= b))
							out_pix[15:8] = (r + b)/2;
						if((g <= r && r <= b) || (b <= r && r <= g))
							out_pix[15:8] = (g + b)/2;
						if((r <= b && b <= g) || (g <= b && b <= r))
							out_pix[15:8] = (r + g)/2;
						out_we = 1;
						next_state = 9;
					 end
					 9: begin //trec la urmatorul pixel si repet procedura
						out_we = 0;
						if(row == 63 && col == 63) begin
							gray_done = 1;
							next_state = 10;
						end
						else begin
							if (row < 63) begin
								if (col < 63) 
									next_col = col + 1;
								else begin
									next_row = row + 1;
									next_col = 0;
								end
							end
							else next_col = col + 1;
							next_state = 8;
						end
					end
					
/*

Fie o matrice 3x3 pentru care notam pozitiile elementelor de la 1 la 9 astfel:

1 2 3
4 5 6
7 8 9

Pixelul de pe pozitia 5 este cel pe care il prelucram la un moment de timp.
De exemplu, daca pixelul curent este pe pozitia (0,0) atunci vecinii lui vor fi pe pozitiile 8, 9 si 6.
Voi parcurge vecinii pixelului in sens trigonometric (3, 2, 1, 4, 7, 8, 9, 6)

*/
					
					10: begin //incep sharpness cu initializarile sale
						next_row = 0;
						next_col = 0;
						filter_done = 0;
						out_we = 0;
						next_state = 11;
					end
					11: begin //prelucrez pixelul din mijloc si apoi verific pe ce linie ma aflu
						new_pix = in_pix[15:8] * 9;
						if(row == 0) next_state = 12;
						else if(row == 63) next_state = 24;
						else next_state = 20;
					end
					12: begin //linia este 0 si verific pe ce coloana ma aflu
						if(col == 0) begin
							next_row = row + 1;
							next_state = 13; //ma duc pe pozitia 8
						end
						else begin
							next_col = col - 1; 
							next_state = 18; //ma duc pe pozitia 4
						end
					end
					13: begin //ma aflu pe pozitia 8
						new_pix = old_pix - in_pix[15:8];
						if (col == 63) begin
							next_row = row - 1;
							next_state = 16; //ma duc pe pozitia 5 (ma intorc de unde am plecat)
						end
						else begin
							next_col = col + 1;
							next_state = 14; //ma duc pe pozitia 9
						end
					end
					14: begin //ma aflu pe pozitia 9
						next_row = row - 1;
						new_pix = old_pix - in_pix[15:8];
						next_state = 15; //ma duc pe pozitia 6
					end
					15: begin //ma aflu pe pozitia 6
						new_pix = old_pix - in_pix[15:8];
						if (row == 0) begin
							next_col = col - 1;
							next_state = 16; //ma duc pe pozitia 5 (ma intorc de unde am plecat)
						end
						else if (row == 63) begin
							next_row = row - 1; //ma duc pe pozitia 3
							next_state = 21;
						end
						else if (row > 0 && col == 1) begin
							next_row = row - 1; //ma duc pe pozitia 3
							next_state = 21;
						end
						else begin
							next_col = col - 1;
							next_state = 16; //ma duc pe pozitia 5 (ma intorc de unde am plecat)
						end
					end
					16: begin //ma aflu pe pozitia 5 (revin de unde am plecat)
						if (old_pix > 255) new_pix = 255;
						if (old_pix < 0) new_pix = 0;
						next_state = 27;
					end
					17: begin //trec la urmatorul pixel
						out_we = 0;
						new_pix = 0;
						if(row == 63 && col == 63) begin
							next_row = 0;
							next_col = 0;
							next_state = 25;
						end
						else begin
							if (row < 63) begin
								if (col < 63) 
									next_col = col + 1;
								else begin
									next_row = row + 1;
									next_col = 0;
								end
							end
							else next_col = col + 1;
							next_state = 11;
						end
					end
					18: begin //ma aflu pe pozitia 4
						new_pix = old_pix - in_pix[15:8];
						if (row == 63) begin
							next_col = col + 1;
							next_state = 16; //ma duc in pe pozitia 5 (ma intorc de unde am plecat)
						end
						else begin
							next_row = row + 1;
							next_state = 19; //ma duc pe pozitia 7
						end
					end
					19: begin //ma aflu pe pozitia 7
						next_col = col + 1; 
						new_pix = old_pix - in_pix[15:8];
						next_state = 13; //ma duc pe pozitia 8
					end
					20: begin //linia este de mijloc si verific pe ce coloana ma aflu
						if(col == 0) begin
							next_row = row + 1;
							next_state = 13; //ma duc pe pozitia 8
						end
						else if(col == 63) begin
							next_row = row - 1;
							next_state = 22; //ma duc pe pozitia 2
						end
						else begin
							next_row = row - 1; 
							next_col = col + 1;
							next_state = 21; //ma duc pe pozitia 3
						end
					end
					21: begin //ma aflu pe pozitia 3
						next_col = col - 1;
						new_pix = old_pix - in_pix[15:8];
						next_state = 22; //ma duc pe pozitia 2
					end
					22: begin //ma aflu pe pozitia 2
						new_pix = old_pix - in_pix[15:8];
						if (col == 0) begin
							next_row = row + 1;
							next_state = 16; //ma duc pe pozitia 5 (ma intorc de unde am plecat)
						end
						else begin
							next_col = col - 1;
							next_state = 23; //ma duc pe pozitia 1
						end
					end
					23: begin //ma aflu pe pozitia 1
						next_row = row + 1;
						new_pix = old_pix - in_pix[15:8];
						next_state = 18; //ma duc pe pozitia 4
					end
					24: begin //linia este 63 si verific pe ce coloana ma aflu
						if(col == 0) begin
							next_col = col + 1;
							next_state = 15; //ma duc pe pozitia 6
						end
						else if(col == 63) begin
							next_row = row - 1;
							next_state = 22; //ma duc pe pozitia 2
						end
						else begin
							next_col = col + 1;
							next_state = 15; //ma duc pe pozitia 6
						end
					end
					25: begin //las doar partea filtrata in pixel si o mut pe mijloc
						out_pix[15:8] = in_pix[7:0];
						out_pix[7:0] = 0;
						next_state = 26;
						out_we = 1;
					end
					26: begin //trec la urmatorul pixel
						out_we = 0;
						if(row == 63 && col == 63) begin
							filter_done = 1;
							next_state = 0;
						end
						else begin
							if (row < 63) begin
								if (col < 63) 
									next_col = col + 1;
								else begin
									next_row = row + 1;
									next_col = 0;
								end
							end
							else next_col = col + 1;
							next_state = 25;
						end
					end
					27: begin //pun pixelul pe iesire dupa ce am verificat daca e numar pozitiv pe 8 biti
						out_pix[7:0] = old_pix;
						out_we = 1;
						next_state = 17;
					end
			 endcase
	 end
	 
endmodule
